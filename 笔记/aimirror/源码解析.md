## 一、main.py 源码解析

### 第一步：点出核心组件

`main.py` 是整个项目的"中控中心"，负责协调各个模块工作。它包含5个核心组件：

- **配置加载**：读取 `config.yaml` 并初始化配置
- **生命周期管理**：服务启动和停止时的资源管理
- **HTTP代理**：将请求转发到上游服务器
- **并行下载**：使用多线程下载大文件
- **请求入口**：接收客户端请求并分发到对应处理函数

### 第二步：用图展示代码流程

```python
def load_handler(handler_path: str):
    """动态加载 handler 模块
    
    Args:
        handler_path: 模块路径，如 "handlers.docker"
    
    Returns:
        模块对象，如果加载失败返回 None
    """
    try:
        module = importlib.import_module(handler_path)
        if not hasattr(module, 'exec_path'):
            logging.error(f"Handler {handler_path} does not have exec_path function")
            return None
        return module
    except ImportError as e:
        logging.error(f"Failed to load handler {handler_path}: {e}")
        return None
```

**解释**：

- **handler_path**：模块路径字符串，如 `"handlers.docker"`
- **exec_path**：Handler 必须提供的入口函数名

```python
async def lifespan(app: FastAPI):

    router = Router(config['rules'])
    cache = CacheManager(

        config['cache']['max_size_gb']
    )
    
    # 初始化全局下载并发控制信号量
    max_concurrent_downloads = config.get('server', {}).get('max_concurrent_downloads', 10)
    download_semaphore = asyncio.Semaphore(max_concurrent_downloads)
    logging.info(f"Global download concurrency limit: {max_concurrent_downloads}")
    
    # HTTP 客户端（带代理支持）
    global upstream_proxy
    upstream_proxy = config['server'].get('upstream_proxy')
    if not upstream_proxy:
        upstream_proxy = None
    
    # 配置连接池限制，避免 PoolTimeout
    limits = httpx.Limits(
        max_connections=200,
        max_keepalive_connections=50,
        keepalive_expiry=30.0
    )
    
    http_client = httpx.AsyncClient(
        timeout=httpx.Timeout(300.0, pool=5),
        follow_redirects=True,
        proxy=upstream_proxy,
        limits=limits
    )
    
    logging.info(f"fast_proxy started, upstream_proxy={upstream_proxy}")
    yield
    
    # 清理
    await http_client.aclose()
    logging.info("fast_proxy stopped")
```

#### 6. 直接代理请求

```python
async def _proxy_request(request: Request, target_url: str, rule: Rule = None) -> Response:
    """直接代理转发请求，支持内容改写"""
    headers = {k: v for k, v in request.headers.items()
               if k.lower() not in ['host', 'connection', 'content-length']}

    # HEAD 请求且配置了 head_meta_headers 时，预获取元数据头部
    meta_headers = {}
    if request.method == "HEAD" and rule and rule.head_meta_headers:
        async with httpx.AsyncClient(timeout=httpx.Timeout(30.0), proxy=upstream_proxy) as head_client:
            meta_resp = await head_client.head(target_url, headers=headers, follow_redirects=False)
            meta_headers = {
                k: v for k, v in meta_resp.headers.items()
                if k.lower() in [h.lower() for h in rule.head_meta_headers]
            }

    async with http_client.stream(
        request.method, target_url, headers=headers,
        content=await request.body(),
        follow_redirects=True
    ) as resp:
        content = await resp.aread()
        content_type = resp.headers.get('content-type', '')

        # 根据 Content-Type 改写内容中的 URL
        if rule:
            content = _rewrite_content_urls(content, rule, content_type)

        response_headers = {k: v for k, v in resp.headers.items()
                           if k.lower() not in ['content-length', 'transfer-encoding', 'content-encoding']}

        response_headers.update(meta_headers)

        return Response(
            content=content,
            status_code=resp.status_code,
            headers=response_headers
        )
```

**解释**：
- **过滤请求头**：移除 `host`、`connection`、`content-length`，避免冲突
- **HEAD 元数据**：预获取额外的响应头（如 HuggingFace 的 `x-repo-commit`）
- **内容改写**：将响应内容中的上游 URL 替换为代理 URL
- **流式读取**：使用 `stream` 和 `aread` 逐步读取响应

#### 7. 内容改写

```python
def _rewrite_content_urls(content: bytes, rule: Rule, content_type: str) -> bytes:
    """改写响应内容中的上游 URL 为代理 URL
    
    Args:
        content: 响应内容
        rule: 匹配的规则
        content_type: 响应的 Content-Type
    """
    if not rule.content_rewrite:
        return content
    
    content_types = rule.content_rewrite.get('content_types', [])
    if not any(ct in content_type for ct in content_types):
        return content
    
    targets = rule.content_rewrite.get('targets', [])
    if not targets:
        return content
    
    try:
        text = content.decode('utf-8')
        
        proxy_host = config['server'].get('public_host', f"127.0.0.1:{config['server']['port']}")
        proxy_base = f"http://{proxy_host}"
        
        for target in targets:
            text = text.replace(target, proxy_base)
        
        return text.encode('utf-8')
    except Exception as e:
        logging.warning(f"Failed to rewrite content URLs: {e}")
        return content
```

**解释**：
- **content_rewrite**：配置中定义的改写规则
- **targets**：要替换的目标 URL 列表
- **public_host**：代理的对外访问地址
- **编码处理**：将字节解码为字符串，替换后再编码

#### 8. 并行下载（核心函数）

```python
async def _parallel_download(request: Request, target_url: str, rule) -> Response:
    """并行下载 + 缓存策略（带全局并发控制和流式返回）"""
    logging.info(f"Entering _parallel_download for: {target_url}")
    
    # 1. 先 HEAD 获取文件大小和内容类型
    headers = {}
    auth_header = request.headers.get('authorization')
    if auth_header:
        headers['Authorization'] = auth_header
    
    async with httpx.AsyncClient(timeout=httpx.Timeout(30.0), proxy=upstream_proxy) as head_client:
        head_resp = await head_client.head(target_url, headers=headers, follow_redirects=True)
        final_url = str(head_resp.url)
        content_length = int(head_resp.headers.get('content-length', 0))
        content_type = head_resp.headers.get('content-type', '')
    
    # 检查文件大小是否符合规则
    if content_length < rule.min_size:
        return await _proxy_request(request, target_url, rule)
    
    # 2. 确定缓存 key
    cache_key_source = getattr(rule, 'cache_key_source', 'final')
    cache_key = target_url if cache_key_source == 'original' else final_url
    
    # 3. 检查缓存
    cached = cache.get(cache_key, content_type)
    if cached:
        logging.info(f"Cache HIT: {cache_key}")
        return StreamingResponse(
            _file_iterator(cached),
            media_type=content_type,
            headers={"Content-Length": str(os.path.getsize(cached))}
        )
    
    # 4. 检查是否正在下载中
    if cache_key in active_downloads:
        temp_file, download_task, start_time = active_downloads[cache_key]
        
        # 长轮询等待下载完成（最多20秒）
        max_wait = 20
        poll_interval = 0.5
        poll_count = int(max_wait / poll_interval)
        
        for _ in range(poll_count):
            await asyncio.sleep(poll_interval)
            
            if download_task.done():
                exc = download_task.exception()
                if exc:
                    raise HTTPException(502, f"Download failed: {str(exc)}")
                
                for _ in range(10):
                    cached = cache.get(cache_key, content_type)
                    if cached:
                        return StreamingResponse(...)
                    await asyncio.sleep(0.5)
                break
        
        # 超时：返回302重试
        return Response(status_code=302, headers={"Location": str(request.url), ...})
    
    # 5. 使用全局信号量控制并发，并开始下载
    async with download_semaphore:
        # 再次检查缓存（可能其他线程已完成）
        cached = cache.get(cache_key, content_type)
        if cached:
            return StreamingResponse(...)
        
        url_hash = hashlib.sha256(cache_key.encode()).hexdigest()[:16]
        temp_file = os.path.join(config['cache']['dir'], f"tmp_{url_hash}")
        
        try:
            downloader = ParallelDownloader(...)
            download_task = asyncio.create_task(...)
            start_time = asyncio.get_event_loop().time()
            active_downloads[cache_key] = (temp_file, download_task, start_time)
            
            # 非高负载：等待下载完成（最多10秒）
            for _ in range(20):
                await asyncio.sleep(0.5)
                if download_task.done():
                    # 等待缓存写入
                    ...
            # 下载未完成：返回302重试
            return Response(status_code=302, ...)
        
        except Exception as e:
            # 清理失败的任务
            ...
```

**解释**：
- **HEAD 请求**：先获取文件元信息，避免下载不必要的数据
- **缓存 key 选择**：`original` 用原始 URL，`final` 用重定向后的 URL
- **缓存命中**：直接返回文件流
- **正在下载**：长轮询等待（最多20秒）
- **并发控制**：使用信号量限制同时下载的任务数
- **后台下载**：创建异步任务在后台下载
- **流式返回**：边下载边返回数据

#### 9. 文件迭代器

```python
async def _file_iterator(filepath: str, chunk_size: int = 8192):
    """文件流式迭代器"""
    with open(filepath, 'rb') as f:
        while chunk := f.read(chunk_size):
            yield chunk

async def _streaming_file_iterator(filepath: str, download_task: asyncio.Task, total_size: int, chunk_size: int = 65536):
    """流式文件迭代器 - 边下载边返回数据"""
    last_size = 0
    max_wait = 600
    start_time = asyncio.get_event_loop().time()
    
    with open(filepath, 'rb') as f:
        while last_size < total_size:
            # 检查超时
            if asyncio.get_event_loop().time() - start_time > max_wait:
                raise TimeoutError(f"Download timeout after {max_wait}s")
            
            # 获取当前文件大小
            f.seek(0, 2)
            current_size = f.tell()
            
            # 如果数据还没准备好，等待
            if current_size <= last_size:
                if download_task.done():
                    exc = download_task.exception()
                    if exc:
                        raise exc
                    if current_size < total_size:
                        raise RuntimeError("Download incomplete")
                    break
                await asyncio.sleep(0.05)
                continue
            
            # 读取已准备好的数据
            f.seek(last_size)
            bytes_to_read = min(current_size - last_size, chunk_size)
            chunk = f.read(bytes_to_read)
            
            if chunk:
                yield chunk
                last_size += len(chunk)
        
        # 下载完成，读取所有剩余数据
        f.seek(last_size)
        while chunk := f.read(chunk_size):
            yield chunk
```

**解释**：
- **_file_iterator**：简单的文件流迭代器，读取已存在的文件
- **_streaming_file_iterator**：边下载边返回的流迭代器
  - 检查文件大小变化（后台任务正在写入）
  - 只返回已下载的数据
  - 等待下载完成返回全部数据

```python
@app.api_route("/{full_path:path}", methods=["GET", "HEAD", "POST", "PUT", "DELETE"])
async def proxy_handler(request: Request, full_path: str):
    """统一代理入口"""
    logging.info(f"Received request: {request.method} {full_path}")
    
    # 路由匹配
    rule = router.match('/' + full_path, content_length=None)
    
    # Handler 特殊处理
    if rule and rule.handler:
        handler_module = load_handler(rule.handler)
        if handler_module:
            try:
                handled, response = await handler_module.exec_path(request, full_path, config, http_client)
                if handled:
                    return response
            except Exception as e:
                raise HTTPException(502, f"Handler error: {str(e)}")
    
    if not rule:
        raise HTTPException(404, "No matching rule found")
    
    # 构建目标 URL
    target_url = rule.build_target_url('/' + full_path)
    
    if rule.strategy == 'proxy':
        return await _proxy_request(request, target_url, rule)
    
    if rule.strategy == 'parallel':
        if request.method == 'HEAD':
            return await _proxy_request(request, target_url, rule)
        if request.method != 'GET':
            return await _proxy_request(request, target_url, rule)
        return await _parallel_download(request, target_url, rule)
    
    return await _proxy_request(request, rule)
```

**解释**：
- **api_route**：匹配所有路径的请求
- **methods**：支持多种 HTTP 方法
- **路由匹配**：根据 URL 找到对应的规则
- **Handler**：如果配置了处理器，先调用处理器
- **策略选择**：根据 `strategy` 选择代理或并行下载

#### 1. Rule 数据类

```python
@dataclass
class Rule:
    name: str
    pattern: str
    upstream: str
    strategy: str
    min_size: int = 0
    max_size: Optional[int] = None
    concurrency: int = 4
    chunk_size: int = 5*1024*1024
    cache_key_source: str = 'final'
    path_rewrite: Optional[list] = None
    content_rewrite: Optional[dict] = None
    handler: Optional[str] = None
    head_meta_headers: Optional[list] = None
    
    def __post_init__(self):
        self._regex = re.compile(self.pattern)
    
    def match(self, path: str) -> bool:
        return bool(self._regex.search(path))
    
    def build_target_url(self, path: str) -> str:
        """构建目标 URL，应用路径重写规则"""
        rewritten_path = path
        if self.path_rewrite:
            for rule in self.path_rewrite:
                rewritten_path = rewritten_path.replace(rule['search'], rule['replace'])
        return f"{self.upstream}{rewritten_path}"
```

**解释**：
- **@dataclass**：自动生成 `__init__`、`__repr__` 等方法
- **pattern**：正则表达式字符串，用于匹配 URL
- **strategy**：下载策略，`proxy` 或 `parallel`
- **min_size**：最小文件大小，小于此值使用代理
- **concurrency**：单任务并发数
- **chunk_size**：分片大小，`<=0` 表示自动计算
- **cache_key_source**：缓存键来源，`original` 或 `final`
- **path_rewrite**：路径重写规则列表
- **content_rewrite**：响应内容改写配置
- **handler**：特殊处理模块路径
- **head_meta_headers**：HEAD 请求时额外保留的响应头

#### 2. Router 类

```python
class Router:
    def __init__(self, rules: list):
        self.rules = [Rule(**r) for r in rules]
    
    def match(self, path: str, content_length: Optional[int] = None) -> Optional[Rule]:
        """匹配第一条符合的规则，并检查大小约束"""
        for rule in self.rules:
            if rule.match(path):
                if content_length is not None:
                    if content_length < rule.min_size:
                        continue
                    if rule.max_size and content_length > rule.max_size:
                        continue
                return rule
        return None
    
    def get_default(self) -> Rule:
        """返回默认规则（通常是 proxy）"""
        for rule in self.rules:
            if rule.name == "default":
                return rule
        return self.rules[-1]
```

**解释**：
- **Rule(**r)**：将字典转换为 Rule 对象
- **匹配顺序**：按配置顺序遍历规则，返回第一个匹配的
- **大小检查**：可选的大小约束，用于按文件大小选择策略
- **default**：兜底规则，确保所有请求都有处理方式

```
    │    └─ 无记录: 返回 None
    │
    └─── 更新访问时间（如果命中）

[cache.put(url, filepath)]
    │
    ├─── 计算 SHA256(url)
    │
    ├─── 创建硬链接或复制文件
    │    │
    │    ├─ 硬链接: 同一文件系统内，节省空间
    │    │
    │    └─ 复制: 跨文件系统时使用
    │
    ├─── 更新 SQLite
    │
    └─── 执行 LRU 淘汰（如果需要）
```

### 第三步：代码详细解析

#### 1. 初始化

```python
class CacheManager:
    def __init__(self, cache_dir: str, max_size_gb: float = 100):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(parents=True, exist_ok=True)
        self.max_bytes = int(max_size_gb * 1024**3)
        self.db_path = self.cache_dir / "meta.db"
        self._init_db()
    
    def _init_db(self):
        """初始化 SQLite 元数据表"""
        conn = sqlite3.connect(self.db_path)
        c = conn.cursor()
        c.execute('''
            CREATE TABLE IF NOT EXISTS files (
                digest TEXT PRIMARY KEY,
                filepath TEXT NOT NULL,
                size INTEGER NOT NULL,
                accessed TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                created TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        c.execute('CREATE INDEX IF NOT EXISTS idx_accessed ON files(accessed)')
        conn.commit()
        conn.close()
```

**解释**：
- **缓存目录**：所有缓存文件存储在这里
- **最大大小**：缓存的容量限制（GB 转 字节）
- **SQLite 数据库**：存储文件元数据（哈希、路径、大小、时间）
- **索引**：加速按访问时间查询（LRU 淘汰）

#### 2. 获取缓存

```python
def _get_digest(self, url: str, content_type: str = "") -> str:
    """生成缓存键：url 的 sha256"""
    return hashlib.sha256(url.encode()).hexdigest()

def get(self, url: str, content_type: str = "") -> Optional[str]:
    """获取缓存文件路径，不存在返回 None"""
    digest = self._get_digest(url, content_type)
    conn = sqlite3.connect(self.db_path)
    c = conn.cursor()
    c.execute('SELECT filepath, size FROM files WHERE digest=?', (digest,))
    row = c.fetchone()
    conn.close()
    
    if row and os.path.exists(row[0]):
        # 更新访问时间
        conn = sqlite3.connect(self.db_path)
        conn.execute('UPDATE files SET accessed=CURRENT_TIMESTAMP WHERE digest=?', (digest,))
        conn.commit()
        conn.close()
        return row[0]
    return None
```

**解释**：
- **SHA256**：URL 的哈希值作为缓存键
- **查询**：从 SQLite 查询文件路径
- **文件存在性**：确认文件实际存在（数据库可能有脏数据）
- **更新访问时间**：用于 LRU 淘汰判断

#### 3. 存入缓存

```python
def put(self, url: str, filepath: str, content_type: str = "") -> str:
    """存入缓存，返回 digest"""
    size = os.path.getsize(filepath)
    digest = self._get_digest(url, content_type)
    
    # 目标路径
    target = self.cache_dir / digest
    if not os.path.exists(target):
        try:
            os.link(filepath, target)  # 硬链接，节省空间
        except OSError:
            # 跨文件系统时硬链接失败，使用复制
            shutil.copy2(filepath, target)
    
    conn = sqlite3.connect(self.db_path)
    c = conn.cursor()
    c.execute('''
        INSERT OR REPLACE INTO files (digest, filepath, size, accessed, created)
        VALUES (?, ?, ?, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP)
    ''', (digest, str(target), size))
    conn.commit()
    conn.close()
    
    # 检查并执行 LRU 淘汰
    self._evict_if_needed()
    return digest
```

**解释**：
- **硬链接**：创建指向同一文件的链接，不占用额外空间
- **复制**：跨文件系统时无法创建硬链接，只能复制
- **INSERT OR REPLACE**：如果已存在则更新
- **LRU 淘汰**：存储后检查容量，超过限制则淘汰

#### 4. LRU 淘汰

```python
def _evict_if_needed(self):
    """LRU 淘汰策略"""
    conn = sqlite3.connect(self.db_path)
    c = conn.cursor()
    
    # 计算当前总大小
    c.execute('SELECT SUM(size) FROM files')
    total = c.fetchone()[0] or 0
    
    while total > self.max_bytes:
        # 找出最久未访问的文件
        c.execute('''
            SELECT digest, filepath, size FROM files 
            ORDER BY accessed ASC LIMIT 1
        ''')
        row = c.fetchone()
        if not row:
            break
        digest, filepath, size = row
        
        # 删除文件
        if os.path.exists(filepath):
            os.remove(filepath)
        c.execute('DELETE FROM files WHERE digest=?', (digest,))
        total -= size
    
    conn.commit()
    conn.close()
```

**解释**：
- **循环淘汰**：可能需要删除多个文件才能满足容量限制
- **ORDER BY accessed ASC**：找出最久未访问的文件
- **文件删除**：删除磁盘文件和数据库记录
- **原子操作**：使用事务确保数据一致性

#### 5. 缓存统计

```python
def get_stats(self) -> dict:
    """返回缓存统计"""
    conn = sqlite3.connect(self.db_path)
    c = conn.cursor()
    c.execute('SELECT COUNT(*), SUM(size), MIN(created), MAX(accessed) FROM files')
    count, size, first, last = c.fetchone()
    conn.close()
    return {
        "count": count or 0,
        "size_bytes": size or 0,
        "size_gb": (size or 0) / 1024**3,
        "first_cached": first,
        "last_accessed": last
    }
```

**解释**：
- **统计信息**：文件数量、总大小、首次缓存时间、最后访问时间
- **聚合函数**：使用 SQL 的 COUNT、SUM、MIN、MAX

### 第一步：点出核心组件

### 第二步：用图展示下载流程

#### 1. Chunk 数据类

#### 2. ParallelDownloader 初始化

```python
class ParallelDownloader:
```

**解释**：

#### 3. 获取文件大小

```python
async def _get_file_size(self, session: aiohttp.ClientSession) -> int:
    """获取文件总大小，检查是否支持 Range"""
    headers = dict(self.headers)
    async with session.head(self.url, headers=headers, allow_redirects=True, proxy=self.proxy) as resp:
        if resp.headers.get('Accept-Ranges') != 'bytes':
            raise ValueError(f"Server does not support Range requests: {self.url}")
        return int(resp.headers.get('Content-Length', 0))
```

**解释**：
- **HEAD 请求**：不下载内容，只获取响应头
- **Accept-Ranges**：检查服务器是否支持分片下载
- **Content-Length**：获取文件总大小

#### 4. 分片计算

```python
def _split_chunks(self, total_size: int) -> List[Chunk]:
    """分割下载范围"""
    # 自动模式：当 chunk_size <= 0 时，自动计算 chunk_size = 总大小 / concurrency
    if self.chunk_size <= 0:
        chunk_size = total_size // max((self.concurrency - 10), 1)
        if chunk_size <= 0:
            chunk_size = total_size
    else:
        chunk_size = self.chunk_size
    
    chunks = []
    for start in range(0, total_size, chunk_size):
        end = min(start + chunk_size - 1, total_size - 1)
        chunks.append(Chunk(start=start, end=end))
    return chunks
```

**解释**：
- **自动模式**：`chunk_size = total_size / (concurrency - 10)`
  - 扣除10个线程预留给其他任务
  - 避免分片数量过多
- **固定模式**：使用配置的固定分片大小
- **最后一个分片**：结束位置是文件末尾

#### 5. 分片下载

```python
async def _download_chunk(self, session: aiohttp.ClientSession, chunk: Chunk, sem: asyncio.Semaphore, retry: int = 3):
    """下载单个分片，带重试机制"""
    async with sem:
        headers = dict(self.headers)
        headers['Range'] = f'bytes={chunk.start}-{chunk.end}'
        
        for attempt in range(retry):
            try:
                async with session.get(self.url, headers=headers, proxy=self.proxy) as resp:
                    if resp.status == 206:
                        chunk.data = await resp.read()
                        return
                    elif resp.status == 200 and chunk.start == 0:
                        # 服务器不支持 Range，返回完整内容
                        chunk.data = await resp.read()
                        return
                    else:
                        raise RuntimeError(f"Chunk download failed: {resp.status}")
            except Exception as e:
                if attempt < retry - 1:
                    wait_time = 2 ** attempt  # 指数退避
                    logging.warning(f"Chunk {chunk.start}-{chunk.end} download failed (attempt {attempt + 1}), retrying in {wait_time}s: {e}")
                    await asyncio.sleep(wait_time)
                else:
                    raise RuntimeError(f"Chunk download failed after {retry} attempts: {e}")
```

**解释**：
- **Range 请求**：`Range: bytes=start-end` 告诉服务器只返回指定范围
- **206 状态码**：服务器支持 Range，返回部分内容
- **200 状态码**：服务器不支持 Range，返回完整内容（只有第一个分片可以接受）
- **指数退避**：失败后等待 `2^attempt` 秒再重试

#### 6. 流式下载

```python
async def download_with_streaming(self, cache_key: str, temp_file: str, cache, content_type: str):
    """流式下载 - 边下载边写入文件"""
    try:
        logging.info(f"Starting streaming download: {self.url}")
        
        async with aiohttp.ClientSession() as session:
            # 1. 获取文件大小
            self.total_size = await self._get_file_size(session)
            
            # 2. 预分配文件空间
            async with aiofiles.open(temp_file, 'wb') as f:
                await f.truncate(self.total_size)
            
            # 3. 分割任务
            self.chunks = self._split_chunks(self.total_size)
            
            # 4. 并发下载并实时写入
            sem = asyncio.Semaphore(self.concurrency)
            
            async def download_and_write(chunk: Chunk):
                """下载单个分片并立即写入文件"""
                async with sem:
                    headers = dict(self.headers)
                    headers['Range'] = f'bytes={chunk.start}-{chunk.end}'
                    
                    for attempt in range(3):
                        try:
                            async with session.get(self.url, headers=headers, proxy=self.proxy) as resp:
                                if resp.status == 206:
                                    data = await resp.read()
                                    # 立即写入文件
                                    async with aiofiles.open(temp_file, 'r+b') as f:
                                        await f.seek(chunk.start)
                                        await f.write(data)
                                    return
                                elif resp.status == 200 and chunk.start == 0:
                                    data = await resp.read()
                                    async with aiofiles.open(temp_file, 'r+b') as f:
                                        await f.write(data)
                                    return
                                else:
                                    raise RuntimeError(f"Chunk download failed: {resp.status}")
                        except Exception as e:
                            if attempt < 2:
                                await asyncio.sleep(2 ** attempt)
                            else:
                                raise
            
            # 启动所有下载任务
            tasks = [download_and_write(chunk) for chunk in self.chunks]
            
            # 等待所有下载完成
            for task in asyncio.as_completed(tasks):
                try:
                    await task
                except Exception as e:
                    logging.error(f"Chunk download failed: {e}")
                    failed_chunks.append(e)
            
            # 验证文件完整性
            actual_size = os.path.getsize(temp_file)
            if actual_size != self.total_size:
                raise RuntimeError(f"File size mismatch")
            
            # 存入缓存
            cache.put(cache_key, temp_file, content_type)
    
    except Exception as e:
        logging.error(f"Streaming download failed: {e}")
        if os.path.exists(temp_file):
            os.remove(temp_file)
        raise
    finally:
        # 清理 active_downloads
        import main
        if cache_key in main.active_downloads:
            del main.active_downloads[cache_key]
```

**解释**：
- **预分配空间**：`truncate(total_size)` 创建指定大小的文件
- **实时写入**：下载完一个分片立即写入，不用等所有分片
- **并发控制**：使用 Semaphore 限制并发数
- **文件完整性**：检查下载后的文件大小是否正确
- **清理状态**：完成后从 `active_downloads` 中移除

```python
def _parse_www_authenticate(authenticate_str: str) -> dict:
    """解析 WWW-Authenticate 头
    示例: Bearer realm="https://auth.docker.io/token",service="registry.docker.io"
    """
    matches = re.findall(r'([a-zA-Z]+)="([^"]*)"', authenticate_str)
    return {k: v for k, v in matches}
```

**解释**：
- **正则匹配**：提取键值对，如 `realm="..."`, `service="..."`
- **返回字典**：方便后续使用

```python
async def _handle_docker_auth(request: Request, upstream: str, http_client) -> Response:
    """处理 Docker Registry 认证请求"""
    # 获取上游的认证信息
    upstream_url = f"{upstream}/v2/"
    headers = {}
    auth_header = request.headers.get('authorization')
    if auth_header:
        headers['Authorization'] = auth_header
    
    resp = await http_client.get(upstream_url, headers=headers, follow_redirects=True)
    
    if resp.status_code != 401:
        # 不需要认证，直接返回
        return Response(...)
    
    www_auth = resp.headers.get('www-authenticate')
    if not www_auth:
        return Response(...)
    
    # 解析 WWW-Authenticate
    auth_info = _parse_www_authenticate(www_auth)
    realm = auth_info.get('realm')
    service = auth_info.get('service', '')
    
    # 构建 token URL
    realm_url = urlparse(realm)
    scope = request.query_params.get('scope', '')
    
    query_params = parse_qs(realm_url.query)
    query_params['service'] = [service]
    if scope:
        query_params['scope'] = [scope]
    
    new_query = urlencode(query_params, doseq=True)
    token_url = urlunparse((...))
    
    # 请求 token
    token_resp = await http_client.get(token_url, headers=..., follow_redirects=True)
    
    return Response(content=token_resp.content, ...)
```

**解释**：
- **上游认证**：先向上游请求 `/v2/`，获取认证要求
- **token URL`：构建上游的 token 服务地址
- **代理 token**：转发 token 给客户端

#### 4. 主处理函数

```python
async def exec_path(request: Request, full_path: str, config: dict, http_client) -> tuple:
    """Docker Registry 特殊处理逻辑"""
    public_host = config['server'].get('public_host', ...)
    
    # 获取 docker registry 上游配置
    docker_upstream = None
    for rule in config.get('rules', []):
        if rule.get('name') == "docker-registry":
            docker_upstream = rule.get('upstream')
            break
    
    # 1. 处理 /v2/auth - token 代理
    if full_path == "v2/auth" or full_path.startswith("v2/auth"):
        response = await _handle_docker_auth(request, docker_upstream, http_client)
        return True, response
    
    # 2. 处理 /v2/ - 检查认证并改写 401 响应
    if full_path == "v2/" or full_path == "v2":
        upstream_url = f"{docker_upstream}/v2/"
        headers = {}
        auth_header = request.headers.get('authorization')
        if auth_header:
            headers['Authorization'] = auth_header
        
        resp = await http_client.get(upstream_url, headers=headers, follow_redirects=True)
        
        if resp.status_code == 401:
            # 返回自定义 401，让客户端向我们的 /v2/auth 请求 token
            is_https = request.url.scheme == "https"
            www_auth = _make_www_authenticate_realm(public_host, is_https)
            return True, Response(
                content=b'{"errors":[{"code":"UNAUTHORIZED",...}]}',
                status_code=401,
                headers={'WWW-Authenticate': www_auth}
            )
        
        return True, Response(...)
    
    # 3. 处理 DockerHub library 镜像重定向
    if full_path.startswith("v2/"):
        path_parts = full_path.split("/")
        if len(path_parts) == 4 and path_parts[2] in ['manifests', 'blobs']:
            repo = path_parts[1]
            if "/" not in repo and not repo.startswith("library"):
                # 重定向到 library/ 路径
                new_path = f"/v2/library/{repo}/{path_parts[2]}/{path_parts[3]}"
                return True, Response(
                    status_code=301,
                    headers={"Location": f"{request.url.scheme}://{public_host}{new_path}"}
                )
    
    # 未处理，继续后续流程
    return False, None
```

**解释**：
- **返回元组**：`(handled, response)`
  - `handled=True`：表示已处理，不再走后续流程
  - `handled=False`：表示未处理，继续后续流程
- **官方镜像重定向**：`nginx` -> `library/nginx`
- **认证改写**：让客户端向代理请求 token
