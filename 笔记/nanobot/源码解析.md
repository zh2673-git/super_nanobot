# nanobot 源码解析

## 第一步：点出核心模块概念

nanobot 的源码由 **6 个核心模块** 构成：

- **agent/loop.py**：代理循环——就像人的大脑，处理所有思考和决策
- **agent/context.py**：上下文构建器——就像人的记忆系统，组织信息
- **agent/memory.py**：持久化记忆——就像人的长期记忆库
- **providers/registry.py**：提供商注册表——就像人的知识库接口
- **bus/queue.py**：消息总线——就像人的神经系统
- **agent/tools/**：工具集——就像人的双手

## 第二步：用图构建模块联系


**联系说明**：外部消息通过 channels 进入 bus，AgentLoop 从 bus 获取消息后，调用 ContextBuilder 构建上下文，通过 Provider 调用 LLM，使用 Tools 执行具体操作。

```python
class AgentLoop:
    """

    职责：

    2. 构建上下文（历史、记忆、技能）
    3. 调用 LLM
    4. 执行工具调用

    """

        self.provider = provider

        self.max_iterations = max_iterations
        self.temperature = temperature
        self.max_tokens = max_tokens
        self.memory_window = memory_window
        
        self.context = ContextBuilder(workspace)
        self.sessions = session_manager or SessionManager(workspace)
        self.tools = ToolRegistry()
        self.subagents = SubagentManager(...)
        
        self._mcp_servers = mcp_servers or {}
        self._mcp_stack: AsyncExitStack | None = None
        self._mcp_connected = False
        
        self._register_default_tools()
    
    def _register_default_tools(self) -> None:
        """注册默认工具集"""
        # 文件工具
        allowed_dir = self.workspace if self.restrict_to_workspace else None
        self.tools.register(ReadFileTool(allowed_dir=allowed_dir))
        self.tools.register(WriteFileTool(allowed_dir=allowed_dir))
        self.tools.register(EditFileTool(allowed_dir=allowed_dir))
        self.tools.register(ListDirTool(allowed_dir=allowed_dir))
        
        # Shell 工具
        self.tools.register(ExecTool(
            working_dir=str(self.workspace),
            timeout=self.exec_config.timeout,
            restrict_to_workspace=self.restrict_to_workspace,
        ))
        
        # Web 工具
        self.tools.register(WebSearchTool(api_key=self.brave_api_key))
        self.tools.register(WebFetchTool())
        
        # 消息工具
        self.tools.register(MessageTool(
            send_callback=self.bus.publish_outbound
        ))
        
        # 子代理工具
        self.tools.register(SpawnTool(manager=self.subagents))
        
        # 定时任务工具
        if self.cron_service:
            self.tools.register(CronTool(self.cron_service))
    
    async def run(self) -> None:
        """运行 Agent 循环"""
        self._running = True
        await self._connect_mcp()  # 连接 MCP 服务器
        logger.info("Agent loop started")
        
        while self._running:
            try:
                msg = await asyncio.wait_for(
                    self.bus.consume_inbound(),
                    timeout=1.0
                )
                response = await self._process_message(msg)
                if response:
                    await self.bus.publish_outbound(response)
            except asyncio.TimeoutError:
                continue
```

```python
class ContextBuilder:
    """
    构建 Agent 的上下文（系统提示 + 消息列表）
    
    组装引导文件、记忆、技能、会话历史
    """
    
    BOOTSTRAP_FILES = ["AGENTS.md", "SOUL.md", "USER.md", "TOOLS.md", "IDENTITY.md"]
    
    def __init__(self, workspace: Path):
        self.workspace = workspace
        self.memory = MemoryStore(workspace)
        self.skills = SkillsLoader(workspace)
    
    def build_system_prompt(self, skill_names: list[str] | None = None) -> str:
        """构建系统提示"""
        parts = []
        
        # 1. 核心身份（时间、环境、工作空间）
        parts.append(self._get_identity())
        
        # 2. 引导文件
        bootstrap = self._load_bootstrap_files()
        if bootstrap:
            parts.append(bootstrap)
        
        # 3. 记忆上下文
        memory = self.memory.get_memory_context()
        if memory:
            parts.append(f"# Memory\n\n{memory}")
        
        # 4. 技能（渐进式加载）
        # - always-loaded 技能：完整内容
        always_skills = self.skills.get_always_skills()
        if always_skills:
            parts.append(f"# Active Skills\n\n{...}")
        
        # - available 技能：仅摘要（Agent 用 read_file 加载）
        skills_summary = self.skills.build_skills_summary()
        if skills_summary:
            parts.append(f"# Skills\n\n{skills_summary}")
        
        return "\n\n---\n\n".join(parts)
    
    def _get_identity(self) -> str:
        """获取核心身份部分"""
        from datetime import datetime
        import platform
        
        now = datetime.now().strftime("%Y-%m-%d %H:%M (%A)")
        workspace_path = str(self.workspace.expanduser().resolve())
        system = platform.system()
        runtime = f"{'macOS' if system == 'Darwin' else system} {platform.machine()}, Python {platform.python_version()}"
        
        return f"""# nanobot 🐈

You are nanobot, a helpful AI assistant.

## Current Time
{now}

## Runtime
{runtime}

## Workspace
Your workspace is at: {workspace_path}
- Long-term memory: {workspace_path}/memory/MEMORY.md
- History log: {workspace_path}/memory/HISTORY.md
"""
    
    def build_messages(
        self,
        history: list[dict],
        current_message: str,
        media: list[str] | None = None,
        channel: str | None = None,
        chat_id: str | None = None,
    ) -> list[dict]:
        """构建完整的消息列表"""
        messages = []
        
        # 系统提示
        system_prompt = self.build_system_prompt()
        if channel and chat_id:
            system_prompt += f"\n\n## Current Session\nChannel: {channel}\nChat ID: {chat_id}"
        messages.append({"role": "system", "content": system_prompt})
        
        # 历史消息
        messages.extend(history)
        
        # 当前消息（支持图片）
        user_content = self._build_user_content(current_message, media)
        messages.append({"role": "user", "content": user_content})
        
        return messages
```

```python
class MemoryStore:
    """
    两层记忆系统：
    - MEMORY.md: 长期记忆（事实、偏好）
    - HISTORY.md: 可搜索的历史日志
    
    就像人的记忆：
    - 长期记忆：记住重要的事情
    - 历史日志：可以回溯查找
    """
    
    def __init__(self, workspace: Path):
        self.memory_dir = ensure_dir(workspace / "memory")
        self.memory_file = self.memory_dir / "MEMORY.md"
        self.history_file = self.memory_dir / "HISTORY.md"
    
    def read_long_term(self) -> str:
        """读取长期记忆"""
        if self.memory_file.exists():
            return self.memory_file.read_text(encoding="utf-8")
        return ""
    
    def write_long_term(self, content: str) -> None:
        """写入长期记忆"""
        self.memory_file.write_text(content, encoding="utf-8")
    
    def append_history(self, entry: str) -> None:
        """追加历史记录"""
        with open(self.history_file, "a", encoding="utf-8") as f:
            f.write(entry.rstrip() + "\n\n")
    
    def get_memory_context(self) -> str:
        """获取记忆上下文"""
        long_term = self.read_long_term()
        return f"## Long-term Memory\n{long_term}" if long_term else ""
```

```python
class Tool(ABC):
    """
    工具基类
    
    所有工具必须实现：
    - name: 工具名称
    - description: 工具描述
    - parameters: JSON Schema 参数定义
    - execute: 执行方法
    """
    
    _TYPE_MAP = {
        "string": str,
        "integer": int,
        "number": (int, float),
        "boolean": bool,
        "array": list,
        "object": dict,
    }
    
    @property
    @abstractmethod
    def name(self) -> str:
        """工具名称"""
        pass
    
    @property
    @abstractmethod
    def description(self) -> str:
        """工具描述"""
        pass
    
    @property
    @abstractmethod
    def parameters(self) -> dict[str, Any]:
        """JSON Schema 参数定义"""
        pass
    
    @abstractmethod
    async def execute(self, **kwargs: Any) -> str:
        """执行工具"""
        pass
    
    def validate_params(self, params: dict[str, Any]) -> list[str]:
        """验证参数，返回错误列表"""
        return self._validate(params, self.parameters, "")
    
    def to_schema(self) -> dict[str, Any]:
        """转换为 OpenAI 函数格式"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters,
            }
        }
```

```python
class ToolRegistry:
    """工具注册表，管理所有可用工具"""
    
    def __init__(self):
        self._tools: dict[str, Tool] = {}
    
    def register(self, tool: Tool) -> None:
        self._tools[tool.name] = tool
    
    def get(self, name: str) -> Tool | None:
        return self._tools.get(name)
    
    def get_definitions(self) -> list[dict[str, Any]]:
        """获取所有工具定义（OpenAI 格式）"""
        return [tool.to_schema() for tool in self._tools.values()]
    
    async def execute(self, name: str, params: dict[str, Any]) -> str:
        tool = self._tools.get(name)
        if not tool:
            return f"Error: Tool '{name}' not found"
        
        try:
            errors = tool.validate_params(params)
            if errors:
                return f"Error: {', '.join(errors)}"
            return await tool.execute(**params)
        except Exception as e:
            return f"Error executing {name}: {str(e)}"
```

### 8. 消息总线 (bus/queue.py)

```python
class MessageBus:
    """
    异步消息总线，解耦渠道和 Agent 核心
    
    渠道推送消息到入站队列，Agent 消费并处理
    Agent 推送响应到出站队列，渠道消费并发送
    """
    
    def __init__(self):
        self.inbound: asyncio.Queue[InboundMessage] = asyncio.Queue()
        self.outbound: asyncio.Queue[OutboundMessage] = asyncio.Queue()
        self._outbound_subscribers: dict[str, list[Callable]] = {}
    
    async def publish_inbound(self, msg: InboundMessage) -> None:
        """渠道发布入站消息"""
        await self.inbound.put(msg)
    
    async def consume_inbound(self) -> InboundMessage:
        """Agent 消费入站消息（阻塞）"""
        return await self.inbound.get()
    
    async def publish_outbound(self, msg: OutboundMessage) -> None:
        """Agent 发布出站响应"""
        await self.outbound.put(msg)
    
    def subscribe_outbound(
        self, 
        channel: str, 
        callback: Callable[[OutboundMessage], Awaitable[None]]
    ) -> None:
        """渠道订阅出站消息"""
        if channel not in self._outbound_subscribers:
            self._outbound_subscribers[channel] = []
        self._outbound_subscribers[channel].append(callback)
```

```python
class ChannelManager:
    """
    管理聊天渠道，协调消息路由
    
    职责：
    - 初始化启用的渠道
    - 启动/停止渠道
    - 路由出站消息
    """
    
    def __init__(self, config: Config, bus: MessageBus):
        self.config = config
        self.bus = bus
        self.channels: dict[str, BaseChannel] = {}
        self._init_channels()
    
    def _init_channels(self) -> None:
        """根据配置初始化渠道"""
        if self.config.channels.telegram.enabled:
            self.channels["telegram"] = TelegramChannel(...)
        if self.config.channels.discord.enabled:
            self.channels["discord"] = DiscordChannel(...)
        if self.config.channels.whatsapp.enabled:
            self.channels["whatsapp"] = WhatsAppChannel(...)
        if self.config.channels.feishu.enabled:
            self.channels["feishu"] = FeishuChannel(...)
        if self.config.channels.slack.enabled:
            self.channels["slack"] = SlackChannel(...)
        if self.config.channels.email.enabled:
            self.channels["email"] = EmailChannel(...)
        if self.config.channels.qq.enabled:
            self.channels["qq"] = QQChannel(...)
    
    async def start_all(self) -> None:
        """启动所有渠道"""
        # 启动出站分发器
        self._dispatch_task = asyncio.create_task(self._dispatch_outbound())
        
        # 并行启动所有渠道
        tasks = []
        for name, channel in self.channels.items():
            tasks.append(asyncio.create_task(channel.start()))
        
        await asyncio.gather(*tasks, return_exceptions=True)
    
    async def _dispatch_outbound(self) -> None:
        """分发出站消息"""
        while True:
            try:
                msg = await asyncio.wait_for(
                    self.bus.consume_outbound(),
                    timeout=1.0
                )
                channel = self.channels.get(msg.channel)
                if channel:
                    await channel.send(msg)
            except asyncio.TimeoutError:
                continue
```

```python
class LiteLLMProvider:
    """
    LiteLLM 提供商实现
    
    统一封装多个 LLM 提供商的调用
    """
    
    def __init__(
        self,
        api_key: str | None = None,
        api_base: str | None = None,
        default_model: str = "anthropic/claude-opus-4-5",
        extra_headers: dict | None = None,
        provider_name: str | None = None,
    ):
        self.api_key = api_key
        self.api_base = api_base
        self.default_model = default_model
        self.extra_headers = extra_headers
        self.provider_name = provider_name
    
    def get_default_model(self) -> str:
        return self.default_model
    
    async def chat(
        self,
        messages: list[dict],
        tools: list[dict] | None = None,
        model: str | None = None,
        temperature: float = 0.7,
        max_tokens: int = 4096,
    ) -> ChatResponse:
        """调用 LLM 聊天 API"""
        model_name = model or self.default_model
        
        # 设置环境变量
        self._setup_env()
        
        # 调用 LiteLLM
        response = await litellm.acompletion(
            model=self._prefix_model(model_name),
            messages=messages,
            tools=tools,
            tool_choice="auto" if tools else None,
            temperature=temperature,
            max_tokens=max_tokens,
            api_base=self.api_base,
            extra_headers=self.extra_headers,
        )
        
        return self._parse_response(response)
    
    def _setup_env(self) -> None:
        """根据提供商设置环境变量"""
        from nanobot.providers.registry import find_by_name
        
        if not self.provider_name:
            return
        
        spec = find_by_name(self.provider_name)
        if spec and spec.env_key and self.api_key:
            os.environ[spec.env_key] = self.api_key
            
            # 额外环境变量
            for key, value in spec.env_extras:
                resolved = value.format(
                    api_key=self.api_key,
                    api_base=self.api_base or spec.default_api_base
                )
                os.environ[key] = resolved
    
    def _prefix_model(self, model: str) -> str:
        """添加模型前缀"""
        from nanobot.providers.registry import find_by_name
        
        if not self.provider_name:
            return model
        
        spec = find_by_name(self.provider_name)
        if spec and spec.litellm_prefix:
            # 检查是否已有前缀
            for prefix in spec.skip_prefixes:
                if model.startswith(prefix):
                    return model
            return f"{spec.litellm_prefix}/{model}"
        
        return model
    
    def _parse_response(self, response) -> ChatResponse:
        """解析 LLM 响应"""
        choice = response.choices[0]
        
        tool_calls = []
        if choice.message.tool_calls:
            for tc in choice.message.tool_calls:
                tool_calls.append(ToolCall(
                    id=tc.id,
                    name=tc.function.name,
                    arguments=json.loads(tc.function.arguments),
                ))
        
        # 获取思考内容 (reasoning_content)
        reasoning_content = getattr(choice.message, "reasoning_content", None)
        
        # 获取思考块 (thinking_blocks) - 用于结构化思考过程
        thinking_blocks = getattr(choice.message, "thinking_blocks", None)
        
        return ChatResponse(
            content=choice.message.content,
            tool_calls=tool_calls,
            has_tool_calls=bool(tool_calls),
            reasoning_content=reasoning_content,
            thinking_blocks=thinking_blocks,
        )
```

```python
class CronService:
    """
    定时任务服务
    
    支持三种调度方式：
    - every: 固定间隔
    - cron: cron 表达式
    - at: 一次性任务
    """
    
    def __init__(self, store_path: Path):
        self.store_path = store_path
        self.jobs: list[CronJob] = []
        self.on_job: Callable[[CronJob], Awaitable[str | None]] | None = None
        self._running = False
    
    async def start(self) -> None:
        """启动调度器"""
        self._load_jobs()
        self._running = True
        asyncio.create_task(self._schedule_loop())
    
    async def _schedule_loop(self) -> None:
        """调度循环"""
        while self._running:
            now_ms = int(time.time() * 1000)
            
            for job in self.jobs:
                if not job.enabled:
                    continue
                if job.state.next_run_at_ms and job.state.next_run_at_ms <= now_ms:
                    asyncio.create_task(self._execute_job(job))
            
            await asyncio.sleep(1)
    
    async def _execute_job(self, job: CronJob) -> None:
        """执行任务"""
        try:
            if self.on_job:
                result = await self.on_job(job)
                logger.info(f"Cron job '{job.name}' completed: {result[:100] if result else 'no output'}")
        except Exception as e:
            logger.error(f"Cron job '{job.name}' failed: {e}")
        finally:
            # 更新下次执行时间
            self._update_next_run(job)
    
    def add_job(
        self,
        name: str,
        schedule: CronSchedule,
        message: str,
        deliver: bool = False,
        to: str | None = None,
        channel: str | None = None,
    ) -> CronJob:
        """添加定时任务"""
        job = CronJob(
            id=str(uuid.uuid4())[:8],
            name=name,
            schedule=schedule,
            payload=JobPayload(
                message=message,
                deliver=deliver,
                to=to,
                channel=channel,
            ),
            enabled=True,
            state=JobState(),
        )
        self._update_next_run(job)
        self.jobs.append(job)
        self._save_jobs()
        return job
    
    def _update_next_run(self, job: CronJob) -> None:
        """计算下次执行时间"""
        if job.schedule.kind == "every":
            next_ms = int(time.time() * 1000) + (job.schedule.every_ms or 0)
        elif job.schedule.kind == "cron":
            # 使用 croniter 解析
            from croniter import croniter
            base = datetime.now()
            iter = croniter(job.schedule.expr or "", base)
            next_ms = int(iter.get_next(datetime).timestamp() * 1000)
        else:
            next_ms = job.schedule.at_ms
        
        job.state.next_run_at_ms = next_ms
        job.state.last_run_at_ms = int(time.time() * 1000)
```

```python
class HeartbeatService:
    """
    心跳服务
    
    周期性唤醒 Agent，用于：
    - 检查 HEARTBEAT.md 文件中的任务
    - 执行后台维护任务
    """
    
    def __init__(
        self,
        workspace: Path,
        on_heartbeat: Callable[[str], Awaitable[str]],
        interval_s: int = 1800,  # 30 分钟
        enabled: bool = True,
    ):
        self.workspace = workspace
        self.on_heartbeat = on_heartbeat
        self.interval_s = interval_s
        self.enabled = enabled
        self._running = False
    
    async def start(self) -> None:
        """启动心跳服务"""
        if not self.enabled:
            return
        self._running = True
        asyncio.create_task(self._heartbeat_loop())
        logger.info(f"Heartbeat service started (interval={self.interval_s}s)")
    
    async def _heartbeat_loop(self) -> None:
        """心跳循环"""
        while self._running:
            await asyncio.sleep(self.interval_s)
            
            # 检查 HEARTBEAT.md
            heartbeat_file = self.workspace / "HEARTBEAT.md"
            if heartbeat_file.exists():
                try:
                    prompt = heartbeat_file.read_text(encoding="utf-8").strip()
                    if prompt:
                        result = await self.on_heartbeat(prompt)
                        logger.info(f"Heartbeat executed: {result[:100] if result else 'done'}")
                        # 清空文件
                        heartbeat_file.write_text("")
                except Exception as e:
                    logger.error(f"Heartbeat failed: {e}")
    
    def stop(self) -> None:
        self._running = False
```
