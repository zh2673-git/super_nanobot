# nanobot 运行逻辑

## 第一步：点出核心流程概念

nanobot 的运行流程可以理解为 **5 个阶段**，就像一条生产线：

- **初始化阶段**：加载配置、创建组件——就像工厂开工前的准备工作
- **启动阶段**：启动各服务和渠道——就像生产线启动运转
- **接收阶段**：从渠道接收用户消息——就像原材料进入生产线
- **处理阶段**：构建上下文、调用 LLM、执行工具——就像加工生产
- **响应阶段**：生成回复、发送给用户——就像产品出厂

**联系说明**：初始化阶段准备所有组件，启动阶段让所有服务并行运行。当用户发送消息时，进入接收阶段，消息被转换为标准格式并放入消息队列。处理阶段是核心，AgentLoop 构建上下文、调用 LLM、执行工具（可能循环多次），最终生成回复。响应阶段将回复通过原渠道发回给用户。

### 启动流程代码

```python
# cli/commands.py 中的 gateway 命令

async def run():
    # 1. 启动定时任务服务
    await cron.start()
    
    # 2. 启动心跳服务（每30分钟唤醒）
    await heartbeat.start()
    
    # 3. 并行启动 Agent 循环和所有渠道
    await asyncio.gather(
        agent.run(),        # Agent 主循环
        channels.start_all(), # 所有启用的渠道
    )
```

## Agent 循环

### 核心处理方法

AgentLoop 的核心是 `_process_message` 方法：

```python
# agent/loop.py

async def _process_message(self, msg: InboundMessage) -> OutboundMessage:
    # 1. 获取或创建会话
    session = self.sessions.get_or_create(msg.session_key)
    
    # 2. 处理特殊命令
    if msg.content.strip().lower() == "/new":
        # 归档旧会话，开始新会话
        session.clear()
        return "New session started"
    
    # 3. 记忆整合（如果消息数超过阈值）
    if len(session.messages) > self.memory_window:
        asyncio.create_task(self._consolidate_memory(session))
    
    # 4. 设置工具上下文（用于消息路由）
    self._set_tool_context(msg.channel, msg.chat_id)
    
    # 5. 构建上下文
    initial_messages = self.context.build_messages(
        history=session.get_history(max_messages=self.memory_window),
        current_message=msg.content,
        channel=msg.channel,
        chat_id=msg.chat_id,
    )
    
    # 6. 执行 Agent 循环
    final_content, tools_used = await self._run_agent_loop(initial_messages)
    
    # 7. 保存会话
    session.add_message("user", msg.content)
    session.add_message("assistant", final_content, tools_used=tools_used)
    self.sessions.save(session)
    
    # 8. 返回响应
    return OutboundMessage(
        channel=msg.channel,
        chat_id=msg.chat_id,
        content=final_content,
    )
```

### 工具执行循环

```python
# agent/loop.py

async def _run_agent_loop(self, initial_messages, on_progress=None):
    messages = initial_messages
    iteration = 0
    final_content = None
    tools_used = []
    
    while iteration < self.max_iterations:
        iteration += 1
        
        # 1. 调用 LLM
        response = await self.provider.chat(
            messages=messages,
            tools=self.tools.get_definitions(),
            model=self.model,
            temperature=self.temperature,
            max_tokens=self.max_tokens,
            reasoning_effort=self.reasoning_effort,  # 思考模式
        )
        
        # 2. 检查是否有工具调用
        if response.has_tool_calls:
            # 显示进度
            if on_progress:
                await on_progress(self._tool_hint(response.tool_calls))
            
            # 添加助手消息到历史
            messages = self.context.add_assistant_message(
                messages, response.content, response.tool_calls
            )
            
            # 3. 执行每个工具调用
            for tool_call in response.tool_calls:
                tools_used.append(tool_call.name)
                result = await self.tools.execute(
                    tool_call.name, 
                    tool_call.arguments
                )
                messages = self.context.add_tool_result(
                    messages, tool_call.id, tool_call.name, result
                )
        else:
            # 没有工具调用，返回最终内容
            final_content = response.content
            break
    
    return final_content, tools_used
```

### 思考模式 (reasoning_effort)

nanobot 支持 LLM 的思考模式（Reasoning Mode），允许模型在生成最终回复前进行更深入的推理：

```python
# reasoning_effort 配置选项
reasoning_effort = "low"    # 低强度思考，适合简单任务
reasoning_effort = "medium" # 中等强度思考，平衡性能和深度
reasoning_effort = "high"   # 高强度思考，适合复杂推理任务
# 或者 None/不设置 = 禁用思考模式
```

思考模式的优势：
- **复杂问题处理**：对于需要多步推理的问题，模型可以先"思考"再回答
- **代码生成**：在编写代码前先规划实现思路
- **数学推理**：分步解决数学问题
- **深入分析**：对主题进行更全面的分析

配置示例：
```json
{
  "agents": {
    "defaults": {
      "model": "anthropic/claude-sonnet-4-5",
      "reasoningEffort": "high"
    }
  }
}
```

### 上下文构建

```python
# agent/context.py

def build_system_prompt(self) -> str:
    parts = []
    
    # 1. 核心身份（时间、运行环境、工作空间）
    parts.append(self._get_identity())
    
    # 2. 加载引导文件
    # - AGENTS.md (Agent 角色定义)
    # - SOUL.md (核心价值观)
    # - USER.md (用户信息)
    # - TOOLS.md (可用工具)
    # - IDENTITY.md (身份定义)
    parts.append(self._load_bootstrap_files())
    
    # 3. 记忆上下文
    memory = self.memory.get_memory_context()
    if memory:
        parts.append(f"# Memory\n\n{memory}")
    
    # 4. 技能（渐进式加载）
    # - always-loaded 技能：完整内容
    # - available 技能：仅显示摘要（Agent 用 read_file 加载）
    always_skills = self.skills.get_always_skills()
    if always_skills:
        parts.append(f"# Active Skills\n\n{...}")
    
    return "\n\n---\n\n".join(parts)
```

## 消息总线

### 消息事件定义

```python
# bus/events.py

@dataclass
class InboundMessage:
    """入站消息：用户 → Agent"""
    channel: str           # 渠道类型 (telegram, discord, etc.)
    sender_id: str         # 发送者标识
    chat_id: str           # 聊天 ID（用于回复）
    content: str           # 消息内容
    media: list[str] | None = None  # 媒体文件路径
    metadata: dict | None = None    # 额外信息
    
    @property
    def session_key(self) -> str:
        return f"{self.channel}:{self.chat_id}"

@dataclass
class OutboundMessage:
    """出站消息：Agent → 用户"""
    channel: str           # 目标渠道
    chat_id: str           # 目标聊天 ID
    content: str           # 响应内容
    metadata: dict | None = None
```

### 异步队列实现

```python
# bus/queue.py

class MessageBus:
    def __init__(self):
        self.inbound: asyncio.Queue[InboundMessage] = asyncio.Queue()
        self.outbound: asyncio.Queue[OutboundMessage] = asyncio.Queue()
        self._outbound_subscribers: dict[str, list[Callable]] = {}
    
    async def publish_inbound(self, msg: InboundMessage) -> None:
        """渠道发布消息到 Agent"""
        await self.inbound.put(msg)
    
    async def consume_inbound(self) -> InboundMessage:
        """Agent 消费消息（阻塞等待）"""
        return await self.inbound.get()
    
    async def publish_outbound(self, msg: OutboundMessage) -> None:
        """Agent 发布响应"""
        await self.outbound.put(msg)
    
    def subscribe_outbound(self, channel: str, callback: Callable) -> None:
        """渠道订阅出站消息"""
        if channel not in self._outbound_subscribers:
            self._outbound_subscribers[channel] = []
        self._outbound_subscribers[channel].append(callback)
```

## 渠道管理

### ChannelManager

```python
# channels/manager.py

class ChannelManager:
    def __init__(self, config: Config, bus: MessageBus):
        self.config = config
        self.bus = bus
        self.channels: dict[str, BaseChannel] = {}
        self._init_channels()
    
    def _init_channels(self) -> None:
        """根据配置初始化渠道"""
        # Telegram
        if self.config.channels.telegram.enabled:
            self.channels["telegram"] = TelegramChannel(...)
        
        # Discord
        if self.config.channels.discord.enabled:
            self.channels["discord"] = DiscordChannel(...)
        
        # WhatsApp
        if self.config.channels.whatsapp.enabled:
            self.channels["whatsapp"] = WhatsAppChannel(...)
        
        # Feishu (WebSocket 长连接)
        if self.config.channels.feishu.enabled:
            self.channels["feishu"] = FeishuChannel(...)
        
        # Slack (Socket Mode)
        if self.config.channels.slack.enabled:
            self.channels["slack"] = SlackChannel(...)
        
        # Email (IMAP/SMTP)
        if self.config.channels.email.enabled:
            self.channels["email"] = EmailChannel(...)
        
        # QQ
        if self.config.channels.qq.enabled:
            self.channels["qq"] = QQChannel(...)
    
    async def _dispatch_outbound(self) -> None:
        """分发出站消息到对应渠道"""
        while True:
            msg = await self.bus.consume_outbound()
            channel = self.channels.get(msg.channel)
            if channel:
                await channel.send(msg)
```

## 工具执行

### ToolRegistry

```python
# agent/tools/registry.py

class ToolRegistry:
    def __init__(self):
        self._tools: dict[str, Tool] = {}
    
    def register(self, tool: Tool) -> None:
        self._tools[tool.name] = tool
    
    def get_definitions(self) -> list[dict]:
        """获取所有工具定义（OpenAI 格式）"""
        return [tool.to_schema() for tool in self._tools.values()]
    
    async def execute(self, name: str, params: dict) -> str:
        tool = self._tools.get(name)
        if not tool:
            return f"Error: Tool '{name}' not found"
        
        # 验证参数
        errors = tool.validate_params(params)
        if errors:
            return f"Error: {', '.join(errors)}"
        
        # 执行工具
        return await tool.execute(**params)
```

```python
async def connect_mcp_servers(
    mcp_servers: dict[str, MCPServerConfig],
    tool_registry: ToolRegistry,
    stack: AsyncExitStack,
) -> None:
    """连接 MCP 服务器并注册工具"""
```
